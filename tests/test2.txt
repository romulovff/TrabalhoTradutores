set copy_set(set s) {
    set ans;
    ans = EMPTY;
    forall(x in s) add(x in ans);
    return ans;
}

set succ(set n) {
    set next;
    next = copy_set(int n);
    return add(n in next);
}

int main() {
    set zero; set one; set two; set three;

    zero = EMPTY;
    one = succ(set zero);
    two = succ(set one);
    three = succ(set two);
}
